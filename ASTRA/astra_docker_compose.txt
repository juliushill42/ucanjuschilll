version: '3.8'

services:
  postgres:
    image: postgres:15-alpine
    container_name: astra-db
    environment:
      POSTGRES_USER: astra_admin
      POSTGRES_PASSWORD: astra_password_2025
      POSTGRES_DB: astra_platform
    ports:
      - "5432:5432"
    volumes:
      - ./init-db:/docker-entrypoint-initdb.d
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U astra_admin"]
      interval: 5s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    container_name: astra-cache
    ports:
      - "6379:6379"
    command: redis-server --appendonly yes
    volumes:
      - redis_data:/data

  go-api:
    build:
      context: ./go-api
      dockerfile: Dockerfile
    container_name: astra-go-api
    environment:
      DATABASE_URL: postgresql://astra_admin:astra_password_2025@postgres:5432/astra_platform
      JWT_SECRET: astra_jwt_secret_12261958_change_in_production
      GIN_MODE: release
    ports:
      - "8080:8080"
    depends_on:
      postgres:
        condition: service_healthy
    restart: unless-stopped

  ai-worker:
    build:
      context: ./ai-worker
      dockerfile: Dockerfile
    container_name: astra-ai-worker
    environment:
      GO_API_URL: http://go-api:8080
    volumes:
      - /tmp/astra:/tmp/astra
    depends_on:
      - go-api
    restart: unless-stopped

  ingest:
    build:
      context: ./rust-ingest
      dockerfile: Dockerfile
    container_name: astra-ingest
    environment:
      RTMP_BIND: 0.0.0.0:1935
      GO_API_URL: http://go-api:8080
      AI_SOCKET_PATH: /tmp/astra/ai.sock
      HLS_OUTPUT_DIR: /tmp/astra/hls
    ports:
      - "1935:1935"
    volumes:
      - /tmp/astra:/tmp/astra
      - hls_data:/tmp/astra/hls
    depends_on:
      - go-api
      - ai-worker
    restart: unless-stopped

  nginx:
    image: nginx:alpine
    container_name: astra-cdn
    ports:
      - "80:80"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf:ro
      - hls_data:/usr/share/nginx/html/hls:ro
    depends_on:
      - ingest
    restart: unless-stopped

volumes:
  postgres_data:
  redis_data:
  hls_data:
